/*
 * This program prompts the user for a number, checks if its even or odd and prints the result
 */

#include <iostream>

using namespace std;

int main() {
	int number = 0;
	cout << "Please enter a number" << endl;
	// Read a number from keyboard
	cin >> number;
	if (number % 2 == 0) 
		cout << "Number is even" << endl;
	}
	else {
		cout << "Number is odd" << endl;
	}
	system("pause");
	return (0);
}
