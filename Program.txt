#include <iostream>

using namespace std;

int main() {
	int a = 2, b = 3, c = 4;
	int result = a + b * c; 
	cout << "Result: " << result << endl;

	int result2 = (2 + 3) * 4; 
	cout << "Result: " << result2 << endl;


	int sum = 55, count = 10;
	cout << "Output: " << sum / count << endl; 
	cout << "Output: " << (double) sum / count << endl; 

	double d1 = sum / count;
	cout << "Output: " << d1 << endl;

	double v = 55;
	double d2 = v / count; 
	cout << "Output: " << d2 << endl; 

	double num = 9;
	cout << sqrt(-num) << endl;

	system("pause");
	return(0);
}
